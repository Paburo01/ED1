void numeranodos(TNodo **elementos){
    TNodo *colocarc=*elementos;
    TNodo *nuevonodo;
    int i=1;
    char cad[10];
    while(colocarc!=NULL){
        nuevonodo=new TNodo;
        itoa(i,cad,10);
        nuevonodo->dato=cad;
        nuevonodo->sig=colocarc->sig;
        colocarc->sig=nuevonodo;
        i++;
        colocarc=colocarc->sig;
    }

};
