void examen::volcar(){
    Tnodo *pos=elementos;
    delete [] tabla;
    n=0;
    
    while(pos->siguiente!=elementos){
            n++;
            pos=pos->siguiente;
    }
    
    n++;//aumento uno mas para que tenga en cuenta el ultimo
    pos=pos->siguiente;//vuelvo al principio
    
    for(int i=0;i<n;i++){
        tabla[i]=pos->dato;
        elementos=elementos->siguiente;
        delete pos;
        pos=elementos;
        
    }
    elementos=NULL;
};
