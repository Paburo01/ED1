void examen::volcar(){
    n=0;
    Tnodo *aux=elementos;
    if(tabla!=NULL){
        delete [] tabla;
        tabla=NULL;
    }else{
        if(aux!=NULL){
            while(aux->siguiente!=elementos){
                aux=aux->siguiente;
                n++;
            }
            n++;
            aux=elementos;
    
            tabla=new [n] int;
            for(int i=0;i<n;i++){
                tabla[i]=aux->dato;
                elementos=elementos->siguiente;
                delete aux;
                aux=elementos;
            }
            elementos=NULL;
        }else{
            cout<<"No hay nodos"<<endl;
        }

    }

};
